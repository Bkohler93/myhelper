Rewrite `cmd/inspect.go` as a full web search diagnostic dry-run. The command runs the search pipeline (gate → fetch → re-rank → build block) in diagnostic mode, printing each stage's output so the user can see exactly what would happen for a given query — without sending any message to the chat model.

Purpose: Make the search pipeline observable. When the gate makes a surprising decision, or re-rank drops useful results, the user can run `inspect` to see exactly why.

Output: A complete `cmd/inspect.go` implementing `runInspect` with section headers, gate output, fetch output, re-rank groups, injected block preview, and token cost line.

Required behavior by flag and gate state:

1. **--no-search** (searchSuppress == true): Print "search suppressed" and return nil immediately. No LLM call, no network call.

2. **--search** (searchForce == true): Print the `---\n[Gate Decision]` header, then `Gate: BYPASSED (--search flag)`. Skip the ollama.Chat gate call entirely. Proceed to fetch.

3. **Gate call** (neither flag): Print `Calling gate LLM...` (plain stdout label, no spinner). Call `ollama.Chat` directly with:
   ```go
   messages := []history.Message{{Role: "user", Content: searchGatePrompt + query}}
   ```
   Print `Gate: YES` or `Gate: NO` (check: `strings.HasPrefix(strings.ToLower(strings.TrimSpace(response)), "yes")`). Print the full trimmed raw response on the next line.
   If gate says NO: print "search not needed" and return nil.
   If gate errors: treat as NO (fail-open) — print "Gate: NO (error: ...)" and "search not needed", return nil.

4. **Fetch** (after gate YES or BYPASSED): Print `---\n[Fetched Results]` header. Print `Fetching results...`. Call `search.Search(query, searchCfg)`. On error or empty results: print the error/empty message and return nil. For each result, print:
   ```
   [N] Title
   URL
   Snippet
   
   ```
   (blank line between entries — matches buildWebBlock entry format)

5. **Re-rank** (always, even when --search forced — per INSP-06): Print `---\n[Re-rank]` header. Print `Calling re-rank LLM...`. Call `reRankResults(query, fetched, cfg)`. Compute dropped = fetched results not in survivors (use URL as identity key). Print:
   - `Survivors (N):` followed by each survivor using same `[N] Title\nURL\nSnippet\n` format
   - `Dropped (N):` followed by each dropped result using same format
   - If survivors == nil (RANK-03 zero case): print `Survivors (0): none` + `Dropped (N): all dropped` then print "injection skipped" and return nil
   - If reRankResults returned all results (RANK-02 error fallback): survivors = all fetched, dropped = 0. Proceed normally.

6. **Injected block preview** (when survivors not nil): Print `---\n[Injected Block]` header. Call `buildWebBlock(survivors, cfg.TokenThreshold/4, cfg)`. If block is empty: print "(no results fit in token budget)" and return nil. Print the full block verbatim. Count tokens:
   ```go
   tokenCost := history.New(cfg.TokenThreshold, []history.Message{{Role: "user", Content: block}}).TokenCount()
   ```
   Print `Token cost: N tokens` (where N is the integer token count).

**RunE function structure**:
```go
func runInspect(cmd *cobra.Command, args []string) error {
    query := args[0]
    cfg := config.Load()
    ApplyFlagOverrides(&cfg)
    searchCfg := search.LoadConfig()

    // --no-search: INSP-07
    if searchSuppress {
        fmt.Println("search suppressed")
        return nil
    }

    // Gate section
    fmt.Println("---")
    fmt.Println("[Gate Decision]")

    var doSearch bool
    if searchForce {
        fmt.Println("Gate: BYPASSED (--search flag)")
        doSearch = true
    } else {
        fmt.Print("Calling gate LLM...")
        messages := []history.Message{{Role: "user", Content: searchGatePrompt + query}}
        response, err := ollama.Chat(cfg, messages)
        fmt.Println() // newline after label
        if err != nil {
            fmt.Printf("Gate: NO (error: %v)\n", err)
            fmt.Println("search not needed")
            return nil
        }
        trimmed := strings.TrimSpace(response)
        doSearch = strings.HasPrefix(strings.ToLower(trimmed), "yes")
        if doSearch {
            fmt.Println("Gate: YES")
        } else {
            fmt.Println("Gate: NO")
        }
        fmt.Println(trimmed)
        if !doSearch {
            fmt.Println("search not needed")
            return nil
        }
    }

    // Fetch section
    fmt.Println("---")
    fmt.Println("[Fetched Results]")
    fmt.Println("Fetching results...")
    fetched, err := search.Search(query, searchCfg)
    if err != nil {
        fmt.Printf("search error: %v\n", err)
        return nil
    }
    if len(fetched) == 0 {
        fmt.Println("(no results returned)")
        return nil
    }
    for i, r := range fetched {
        fmt.Printf("[%d] %s\n%s\n%s\n\n", i+1, r.Title, r.URL, r.Snippet)
    }

    // Re-rank section
    fmt.Println("---")
    fmt.Println("[Re-rank]")
    fmt.Print("Calling re-rank LLM...")
    survivors, _ := reRankResults(query, fetched, cfg)
    fmt.Println() // newline after label

    // Compute dropped by URL identity
    survivorURLs := make(map[string]bool)
    for _, r := range survivors {
        survivorURLs[r.URL] = true
    }
    var dropped []search.Result
    for _, r := range fetched {
        if !survivorURLs[r.URL] {
            dropped = append(dropped, r)
        }
    }

    if survivors == nil {
        fmt.Printf("Survivors (0): none\n")
        fmt.Printf("Dropped (%d): all dropped\n", len(fetched))
        fmt.Println("injection skipped")
        return nil
    }

    fmt.Printf("Survivors (%d):\n", len(survivors))
    for i, r := range survivors {
        fmt.Printf("[%d] %s\n%s\n%s\n\n", i+1, r.Title, r.URL, r.Snippet)
    }
    fmt.Printf("Dropped (%d):\n", len(dropped))
    for i, r := range dropped {
        fmt.Printf("[%d] %s\n%s\n%s\n\n", i+1, r.Title, r.URL, r.Snippet)
    }

    // Injected block section
    fmt.Println("---")
    fmt.Println("[Injected Block]")
    budget := cfg.TokenThreshold / 4
    block := buildWebBlock(survivors, budget, cfg)
    if block == "" {
        fmt.Println("(no results fit in token budget)")
        return nil
    }
    fmt.Print(block)
    tokenCost := history.New(cfg.TokenThreshold, []history.Message{{Role: "user", Content: block}}).TokenCount()
    fmt.Printf("Token cost: %d tokens\n", tokenCost)

    return nil
}
```
