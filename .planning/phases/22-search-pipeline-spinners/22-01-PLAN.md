```go
type spinner struct{ stop chan struct{} }

func startSpinner(label string) *spinner {
	s := &spinner{stop: make(chan struct{})}
	go func() {
		frames := []rune{'|', '/', '-', '\\'}
		i := 0
		t := time.NewTicker(100 * time.Millisecond)
		defer t.Stop()
		for {
			fmt.Fprintf(os.Stderr, "\r%c %s", frames[i], label)
			i = (i + 1) % len(frames)
			select {
			case <-s.stop:
				fmt.Fprintf(os.Stderr, "\r%s\r", strings.Repeat(" ", len(label)+3))
				return
			case <-t.C:
			}
		}
	}()
	return s
}

func (s *spinner) done() { close(s.stop) }
```

No Bubble Tea, no charmbracelet packages, no external dependencies. Uses only `fmt`, `os`, `strings`, `time` — all stdlib.
