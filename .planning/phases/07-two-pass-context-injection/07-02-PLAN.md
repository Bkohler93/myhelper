func runXxx(cmd *cobra.Command, args []string) error {
    input, err := resolveInput(args, "prompt: ")
    if err != nil { return err }

    projectCtx, err := appctx.LoadContext()
    if err != nil { return err }

    cfg := config.Load()

    root, err := os.Getwd()
    if err != nil {
        return fmt.Errorf("runXxx: getwd: %w", err)
    }
    injected, err := buildInjectedMessages(root, input, cfg, ollama.Chat, pass1XxxFocus)
    if err != nil {
        return fmt.Errorf("runXxx: buildInjectedMessages: %w", err)
    }

    messages := []history.Message{
        {Role: "system", Content: buildSystemMessage(projectCtx, xxxSystemPrompt)},
    }
    messages = append(messages, injected...)

    h := history.New(cfg.TokenThreshold, messages)

    err = initiateConversation(cfg, h, ollama.StreamChat)
    ...
}
