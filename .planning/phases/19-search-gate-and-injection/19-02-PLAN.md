    ```go
    // buildUserMessage augments the user query with a [WEB RESULTS] block if appropriate.
    // Returns the original query unchanged when search is skipped.
    // noSearch (GATE-04) takes priority over forceSearch (GATE-03).
    func buildUserMessage(query string, cfg config.Config, searchCfg search.Config, forceSearch, noSearch bool) string {
        if noSearch {
            return query // GATE-04: --no-search suppresses all search
        }

        doSearch := forceSearch // GATE-03: --search forces gate=true
        if !doSearch {
            doSearch = searchGate(query, cfg) // GATE-01/GATE-02
        }
        if !doSearch {
            return query
        }

        results, err := search.Search(query, searchCfg)
        if err != nil || len(results) == 0 {
            return query // network/empty: degrade gracefully
        }

        ranked, _ := reRankResults(query, results, cfg)
        if ranked == nil {
            return query // RANK-03: zero relevant results → skip injection
        }

        budget := cfg.TokenThreshold / 4 // 25% reserved for web context
        block := buildWebBlock(ranked, budget, cfg)
        if block == "" {
            return query
        }
        return block + query // block BEFORE query (context before question, per INJ-01)
    }
    ```

| Boundary | Description |
|----------|-------------|
| CLI flags → search behavior | --search/--no-search flags control whether network calls are made |
| buildUserMessage output → hist.Add | Augmented message (potentially containing web snippets) added to LLM history |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-19-05 | Tampering | --search/--no-search flag parsing | accept | Cobra BoolVar; no string parsing; boolean cannot carry injection payload |
| T-19-06 | Spoofing | Web snippet content in REPL history | accept | buildUserMessage wraps content in `[WEB RESULTS]` delimited block; local Ollama model; no external tool-use surface |
| T-19-07 | Denial of Service | Per-REPL-turn search calls | accept | Gate LLM call and SearXNG fetch add latency per turn when search fires; acceptable for local deployment; user can use --no-search to disable |
| T-19-08 | Repudiation | search.LoadConfig called once at startup | accept | LoadConfig reads env/files at startup; if env var changes mid-session the new value is not picked up — this is intentional (Pitfall 6 mitigation); no security implication |
</threat_model>
