```go
func runPlan(cmd *cobra.Command, args []string) error {
    input, err := resolveInput(args, "Describe the feature or task: ")
    if err != nil {
        return err
    }

    projectCtx, err := appctx.LoadContext()
    if err != nil {
        return err
    }

    cfg := config.Load()

    // Build history with system message + first user turn
    hist := history.New(cfg.TokenThreshold)
    hist.Add("system", buildSystemMessage(projectCtx, planSystemPrompt))
    hist.Add("user", input)

    // First turn — stream the initial response
    response, err := ollama.StreamChat(cfg, hist.Messages())
    if err != nil {
        return err
    }
    hist.Add("assistant", response)

    // Hand off to the shared conversation loop for follow-up turns
    return runConversationLoop(cfg, hist, ollama.StreamChat)
}
```
