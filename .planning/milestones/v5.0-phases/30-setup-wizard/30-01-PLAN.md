detectMemoryMiB() platform branching:
```
switch runtime.GOOS {
case "linux":
    // try nvidia-smi first
    out, err := exec.Command("nvidia-smi", "--query-gpu=memory.total", "--format=csv,noheader,nounits").Output()
    if err == nil && len(strings.TrimSpace(string(out))) > 0 {
        if n, err := strconv.ParseInt(strings.TrimSpace(string(out)), 10, 64); err == nil {
            return n
        }
    }
    // fallback: /proc/meminfo
    data, err := os.ReadFile("/proc/meminfo")
    if err == nil {
        for _, line := range strings.Split(string(data), "\n") {
            if strings.HasPrefix(line, "MemTotal:") {
                fields := strings.Fields(line)
                if len(fields) >= 2 {
                    if kb, err := strconv.ParseInt(fields[1], 10, 64); err == nil {
                        return kb / 1024
                    }
                }
            }
        }
    }
    return 0
case "darwin":
    out, err := exec.Command("system_profiler", "SPHardwareDataType").Output()
    if err == nil {
        for _, line := range strings.Split(string(out), "\n") {
            if strings.Contains(line, "Memory:") {
                parts := strings.Fields(strings.TrimSpace(line))
                // parts: ["Memory:", "32", "GB"]
                if len(parts) >= 3 {
                    if n, err := strconv.ParseInt(parts[1], 10, 64); err == nil {
                        unit := strings.ToUpper(parts[2])
                        if unit == "GB" { return n * 1024 }
                        if unit == "MB" { return n }
                    }
                }
            }
        }
    }
    // sysctl fallback
    out, err = exec.Command("sysctl", "-n", "hw.memsize").Output()
    if err == nil {
        if n, err := strconv.ParseInt(strings.TrimSpace(string(out)), 10, 64); err == nil {
            return n / 1024 / 1024
        }
    }
    return 0
default:
    return 0
}
```

pullModel() must use a bufio.Scanner over resp.Body (HTTP response body, not the wizard's stdin reader — this is safe because it wraps a separate io.Reader):
```
body, _ := json.Marshal(pullRequest{Name: name, Stream: true})
resp, err := http.Post("http://localhost:11434/api/pull", "application/json", bytes.NewReader(body))
if err != nil { return fmt.Errorf("pull request: %w", err) }
defer resp.Body.Close()
sc := bufio.NewScanner(resp.Body)
for sc.Scan() {
    var p pullProgress
    if err := json.Unmarshal(sc.Bytes(), &p); err != nil { continue }
    if p.Error != "" { return fmt.Errorf("pull error: %s", p.Error) }
    if p.Total > 0 {
        fmt.Fprintf(w, "\r%s %.0f%%  ", p.Status, float64(p.Completed)/float64(p.Total)*100)
    } else {
        fmt.Fprintf(w, "\r%s       ", p.Status)
    }
    if p.Status == "success" { fmt.Fprintln(w); break }
}
return sc.Err()
```

Note: bufio.Scanner on resp.Body is fine here — it wraps a separate HTTP response body io.Reader, not the wizard's stdin io.Reader. The pitfall about multiple scanners only applies to wrapping the same underlying reader twice.

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-30-01 | Information Disclosure | mergeHomeConfig / os.WriteFile | mitigate | Write config with 0600 permissions — owner read/write only; Tavily key never printed to stdout/stderr |
| T-30-02 | Information Disclosure | pullModel / fmt.Fprintf progress | accept | Pull progress prints model status strings, never the API key; key is not in scope at pull stage |
| T-30-03 | Tampering | SearXNG endpoint input | mitigate | Validate that endpoint has http:// or https:// prefix before writing to config; reject bare hostnames |
| T-30-04 | Tampering | Config merge overwrites existing keys | mitigate | Use map[string]interface{} merge (not typed struct re-marshal) — only keys in `updates` map are modified |
| T-30-05 | Denial of Service | Ollama pull hangs indefinitely | accept | http.DefaultClient has no timeout for long-running pull; model pulls can take minutes — this is expected behavior, not a DoS concern for a local CLI |
