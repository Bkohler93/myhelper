    ```go
    func LoadConfig() Config {
        cfg := Config{Endpoint: DefaultSearchEndpoint}

        if loaded, ok := loadConfigFile(homeConfigPath()); ok {
            if loaded.Endpoint != "" {
                cfg.Endpoint = loaded.Endpoint
            }
        }
        if loaded, ok := loadConfigFile(localConfigPath()); ok {
            if loaded.Endpoint != "" {
                cfg.Endpoint = loaded.Endpoint
            }
        }

        if v := os.Getenv("MYHELPER_SEARCH_ENDPOINT"); v != "" {
            cfg.Endpoint = v
        }
        return cfg
    }

    func localConfigPath() string { return ".myhelper/config.json" }

    func homeConfigPath() string {
        home, err := os.UserHomeDir()
        if err != nil {
            return ""
        }
        return filepath.Join(home, ".config", "myhelper", "config.json")
    }

    func loadConfigFile(path string) (Config, bool) {
        if path == "" {
            return Config{}, false
        }
        data, err := os.ReadFile(path)
        if err != nil {
            return Config{}, false
        }
        var c Config
        if err := json.Unmarshal(data, &c); err != nil {
            return Config{}, false
        }
        return c, true
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| CLI user → search.Search() | `query` parameter originates from user input at the terminal |
| search.Search() → SearXNG HTTP endpoint | Outbound GET request; endpoint is operator-configured (env/config file), not user-supplied |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-18-01 | Tampering | `search.go Search()` URL construction | mitigate | Use `url.QueryEscape(query)` for the `q=` parameter — prevents query injection via special chars (`&`, `=`, `#`, space) that could alter the request structure (V5 Input Validation) |
| T-18-02 | Elevation of Privilege | `search.go LoadConfig()` SSRF via endpoint value | accept | Endpoint is set by the operator via env var or config file on the local machine — not by end users. SSRF risk is inherent in any configurable HTTP endpoint for a local CLI tool; the threat model accepts this. The `http://127.0.0.1:1` test confirms unreachable endpoints return errors cleanly. |
| T-18-03 | Information Disclosure | `search.go Search()` error messages | accept | Error messages include the request URL and HTTP status. For a local CLI tool running as the operator, this is diagnostic information, not a leakage risk. |
</threat_model>
